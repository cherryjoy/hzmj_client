RoleInfoCell = {}

function RoleInfoCell:SetData(roleId,roleName)
	self.roleId=roleId
	self.roleName=roleName
	self.roleNameLbl:set_text(roleName)
end

function RoleInfoCell:RoleInfoCell_OnClick( )
	RoleInfoCell.SetSelectState(self)
	if self.roleId~=-1 then
		ChooseRoleUI.SetData(self.chooseRoleSelf,self.roleId,self.roleName)
	else
		self.chooseRoleSelf.createRole:SetActive(true)
		self.chooseRoleSelf.gameObject:SetActive(false)
	end
end

function RoleInfoCell:ClearData( )
	self.roleNameLbl:set_text("添加")
	self.roleId=-1
	self.roleName=""
end

function RoleInfoCell:ClearSelectState( )
	self.selectedObj:SetActive(false)
end

function RoleInfoCell:SetSelectState( )
	ChooseRoleUI.ClearSelectStates(self.chooseRoleSelf)
	self.selectedObj:SetActive(true)
end
